 16 
 17 			var e = (this.jquery) ? this[0] : this;
 18 
 19 			return (
 20 
 21 				/* mozilla / dom 3.0 */
 22 				('selectionStart' in e && function() {
 23 					var l = e.selectionEnd - e.selectionStart;
 24 					return { start: e.selectionStart, end: e.selectionEnd, length: l, text: e.value.substr(e.selectionStart, l) };
 25 				}) ||
 26 
 27 				/* exploder */
 28 				(document.selection && function() {
 29 
 30 					e.focus();
 31 
 32 					var r = document.selection.createRange();
 33 					if (r === null) {
 34 						return { start: 0, end: e.value.length, length: 0 }
 35 					}
 36 
 37 					var re = e.createTextRange();
 38 					var rc = re.duplicate();
 39 					re.moveToBookmark(r.getBookmark());
 40 					rc.setEndPoint('EndToStart', re);
 41 
 42 					return { start: rc.text.length, end: rc.text.length + r.text.length, length: r.text.length, text: r.text };
 43 				}) ||
 44 
 45 				/* browser not supported */
 46 				function() { return null; }
 47 
 48 			)();
 49 
 50 		},
 51 
 52 		replaceSelection: function() {
 53 
 54 			var e = (this.jquery) ? this[0] : this;
 55 			var text = arguments[0] || '';
 56 
 57 			return (
 58 
 59 				/* mozilla / dom 3.0 */
 60 				('selectionStart' in e && function() {
 61 					e.value = e.value.substr(0, e.selectionStart) + text + e.value.substr(e.selectionEnd, e.value.length);
 62 					return this;
 63 				}) ||
 64 
 65 				/* exploder */
 66 				(document.selection && function() {
 67 					e.focus();
 68 					document.selection.createRange().text = text;
 69 					return this;
 70 				}) ||
 71 
 72 				/* browser not supported */
 73 				function() {
 74 					e.value += text;
 75 					return $(e);
 76 				}
 77
